function login() { var username = document.getElementById('username').value.trim().toLowerCase(); var password = document.getElementById('password').value.trim(); // Worker accounts with their passwords and redirect URLs const users = { "admin": { password: "password", redirect: "https://sites.google.com/view/tsg-fun-homepage/home" }, "worker1": { password: "worker1pass", redirect: "https://sites.google.com/view/tsg-worker1-login" }, "worker2": { password: "worker2pass", redirect: "https://sites.google.com/view/tsg-worker2-login" }, "worker3": { password: "worker3pass", redirect: "https://sites.google.com/view/tsg-worker3-login" } }; if (users[username] && users[username].password === password) { window.location.href = users[username].redirect; } else { document.getElementById('error-message').textContent = 'Invalid username or password!'; } }
